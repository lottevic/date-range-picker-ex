...
   
[js]       
$('input[name="NAME_ONE_DATE3"]').daterangepicker({
  singleDatePicker: true,
  minYear: parseInt(moment().format('YYYY'), 10)-10, //起始年
  maxYear: parseInt(moment().format('YYYY'), 10)+1, //結束年
  showDropdowns: true, //年月可用下拉選單選
  showWeekNumbers: true,
  locale: {
    format: 'YYYY/MM/DD',
    weekLabel: "週次",
    daysOfWeek: ["日", "一", "二", "三", "四", "五", "六"],
    monthNames: ["一月", "二月", "三月", "四月", "五月", "六月", "七月", "八月", "九月", "十月", "十一月", "十二月"]
  }
});

[js] 
var start = moment();  
$('#ID_DATE').daterangepicker({ 
  singleDatePicker: true,
  drops: 'up',
}, function (start) {
  $('#ID_DATE span').html(start.format('YYYY/MM/DD'));
});
強制關閉手機input keyboard的寫法
手機適用 
//強制關閉手機input keyboard
$('input.lotte-input-date').on('show.daterangepicker', function(ev, picker) { 
  //當打開小月曆時，
  var myinput = $(this);
  if($('.ID_HIDE_MOBILE_KEYBOARD').prop('checked')){
    myinput.blur(); //將手機keyboard關閉
    myinput.attr('disabled', 'disabled'); //將input設置為disable狀態，(鎖住) 避免再次點擊到而打開手機keyboard
  }
});
$('input.lotte-input-date').on('hide.daterangepicker', function(ev, picker) { 
  //當關閉小月曆時，
  var myinput = $(this);
  myinput.removeAttr('disabled'); //取消input的disable狀態 (解鎖)
});
